# Online Earning Website - Environment Variables (example)
# Copy/rename to .env.local in development and set real values.
# On Vercel, add these in Project Settings > Environment Variables.

# App
NEXT_PUBLIC_SITE_URL=https://weearn.sbs
NEXT_PUBLIC_APP_NAME=WeEarn
NEXT_PUBLIC_SUPABASE_URL=
NEXT_PUBLIC_SUPABASE_ANON_KEY=
# Server-only (never expose in client)
SUPABASE_SERVICE_ROLE_KEY=

# NOWPayments (TRC20 USDT) - Note: use lowercase as required by NOWPayments API
NEXT_PUBLIC_NOWPAYMENTS_CURRENCY=usdttrc20
NOWPAYMENTS_API_KEY=
NOWPAYMENTS_IPN_SECRET=
# Optional base URL override
NOWPAYMENTS_BASE_URL=https://api.nowpayments.io/v1

# Security
# Optionally used for signing custom webhooks or revalidate endpoints
REVALIDATE_SECRET=
# Admin secret for setting user roles (default: admin123)
ADMIN_SECRET=admin123

# Notes
# - REVALIDATE_SECRET must be sent as Authorization: Bearer <REVALIDATE_SECRET>
#   when calling /api/cron/daily-returns
# - SUPABASE_SERVICE_ROLE_KEY is required on the server for webhooks/cron/admin pages.
# - On Vercel, set NEXT_PUBLIC_SITE_URL to your production URL, e.g. https://weearn.sbs
# - To make an account admin, use the Admin Tools in /admin page with ADMIN_SECRET
# - ADMIN_SECRET is used for the /api/admin/set-role endpoint to grant admin access
# - Admin users can top up other users' balances via the Admin Tools
